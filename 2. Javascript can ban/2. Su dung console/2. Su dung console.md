## 1. Sử dụng các câu lệnh console

console.log(123)
console.log(true)

console.log(name)
console.log([1, 2, 3])
console.table({name: 'Duoc', age: 18})
console.warn('Cảnh báo')
console.error('Lỗi rồi')
console.clear()
console.time('start')
console.log(123)

console.timeEnd('start)
